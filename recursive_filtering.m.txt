function u_filtered = recursive_filtering(u, F)
    % RECURSIVE_FILTERING Applies recursive filtering using 1/F
    %
    % Inputs:
    %   u - Input signal (vector)
    %   F - Filter coefficients (vector)
    %
    % Output:
    %   u_filtered - Filtered input signal (vector)
    
    N = length(u);  % Number of data points
    order = length(F) - 1;  % Order of the filter
    u_filtered = zeros(N, 1);  % Initialize filtered signal
    
    for t = 1:N
        % Apply filter recursively
        if t > order
            u_filtered(t) = u(t) - F(2:end)' * u_filtered(t-1:-1:t-order);
        else
            u_filtered(t) = u(t);
        end
    end
end
